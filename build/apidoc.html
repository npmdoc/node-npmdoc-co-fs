function co-fs.appendFile ()

  • description and source-code
    appendFile = function (){
  var args = [].slice.call(arguments);
  var results;
  var called;
  var cb;

  args.push(function(){
    results = arguments;

    if (cb && !called) {
      called = true;
      cb.apply(this, results);
    }
  });

  fn.apply(this, args);

  return function(fn){
    cb = fn;

    if (results && !called) {
      called = true;
      fn.apply(this, results);
    }
  }
}
